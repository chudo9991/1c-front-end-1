----------- √лобальные библиотеки и локальные библиотеки npm ------------

// √де установлены глобальные библиотеки.

	npm list -g

// ѕосмотреть не -g lobal библиотеки в node_modules в той папке, в которой вы находитесь

	npm list

// Ќайти глобальный установочный корень независимо от текущего рабочего каталога

	npm root -g

// —ообщит вам эффективный каталог установки ваших локальных пакетов npm

	npm root

// ѕуть к библиотеке в Windows 7, 8 и 10 - %USERPROFILE%\AppData\Roaming\npm\node_modules
// если не установлен NODE_PATH


------------- ”становка npm модулей ---------------

// ¬ командную строку будет выведен список: текущей версии, требуемой версии и последней версии.

// ѕроверить пакеты, необъ€вленные в package.json
