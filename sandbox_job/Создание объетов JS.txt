// Сначала создадим шаблон со свойствами и функциями (зедесь безе фукции)
// Функця Car будет использоваться в качестве шаблона для создания объекта
// Ключевое слово this что свойсто или метод будет относится к объкту, созданного этой фукцией
// Имя фукции конструктора начинается с заглавной буквы и напоминает , что создание нового объкта при помщи этой функции используется ключевое слово new

// НОТАЦИЯ КОНСТРУКТОРА

function Car(make, model, year, owner) {
  this.make = make;
  this.model = model;
  this.year = year;
  this.owner = owner;
}


// Экзепляры создаются с использованием ФУНКЦИИ КОНСТРУКТОРА
